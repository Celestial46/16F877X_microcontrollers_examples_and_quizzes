#INCLUDE <P16F877A.INC>
SAYAC EQU 0X20
	ORG 0X00
	GOTO MAIN
	ORG 0X04
KESME
	BANKSEL PIR1
	BCF PIR1,TMR1IF
	MOVLW H'3C'
	MOVWF TMR1H
	MOVLW H'B0'
	MOVWF TMR1L
	BANKSEL SAYAC
	DECFSZ SAYAC,F
	RETFIE
	MOVLW D'10'
	MOVWF SAYAC
	INCF PORTB
	RETFIE
MAIN
	BANKSEL ADCON1
	MOVLW D'6'
	MOVWF ADCON1
	BANKSEL TRISB
	MOVLW B'000011'
	CLRF TRISB
	BANKSEL PORTB
	CLRF PORTB
	CLRF PORTA
	BANKSEL INTCON
	BSF INTCON,GIE
	BSF INTCON,PEIE ;?
	BANKSEL PIE1
	BSF PIE1,TMR1IE
	BANKSEL T1CON
	MOVLW B'00010101'
	MOVWF T1CON
	BANKSEL PIR1
	BCF PIR1,TMR1IF
	BANKSEL TMR1H
	MOVLW H'3C'
	MOVWF TMR1H
	MOVLW H'B0'
	MOVWF TMR1L
	MOVLW D'10'
	MOVWF SAYAC
DONGU
	GOTO DONGU
	END
