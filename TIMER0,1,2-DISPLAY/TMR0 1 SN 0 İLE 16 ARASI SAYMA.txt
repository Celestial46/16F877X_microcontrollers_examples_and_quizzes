#INCLUDE <P16F877A.INC>
SAYI EQU 0X20
DEGER EQU 0X21
	ORG 0X00
	GOTO MAIN
	ORG 0X04
	GOTO KESME
	
KESME
	
	BANKSEL INTCON
	BCF INTCON,TMR0IF
	DECFSZ SAYI,F
	RETFIE
	GOTO ISLEM
ISLEM
	DECF DEGER
	BTFSS STATUS,Z
	GOTO YAK
	GOTO SIFIRLA
SIFIRLA
	MOVLW D'16'
	MOVWF SAYI
	MOVWF DEGER
	CLRF PORTB
	RETFIE
YAK
	BANKSEL PORTB
	INCF PORTB
	
	MOVLW D'16'
	MOVWF SAYI
	RETFIE


MAIN
	BANKSEL TRISB
	CLRF TRISB
	BANKSEL PORTB
	CLRF PORTB
	BANKSEL OPTION_REG
	MOVLW B'00000111'
	MOVWF OPTION_REG
	BANKSEL INTCON
	BSF INTCON,GIE
	BSF INTCON,TMR0IE
	BCF INTCON,TMR0IF
	MOVLW D'16'
	MOVWF DEGER
	MOVWF SAYI
DONGU
	GOTO DONGU

	END
