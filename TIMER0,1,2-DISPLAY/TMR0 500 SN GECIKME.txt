list p=16f877
include "p16f877.inc"
SAYAC equ 0x25
	org 0x00
	GOTO MAIN
	ORG 0X04
	GOTO KESME
MAIN
	BANKSEL TRISB
	CLRF TRISB
	BANKSEL PORTB
	CLRF PORTB
	MOVLW D'7'
	MOVWF SAYAC
	BANKSEL OPTION_REG
	MOVLW B'00000111'
	MOVWF OPTION_REG
	BANKSEL INTCON
	
	BSF INTCON,GIE
	BSF INTCON,TMR0IE
	BCF INTCON,TMR0IF
DONGU
	GOTO DONGU
KESME
	BANKSEL INTCON
	BCF INTCON,TMR0IF
	DECFSZ SAYAC,F
	RETFIE
	GOTO ISLEM
ISLEM
	BANKSEL PORTB
	INCF PORTB
	MOVLW D'7'
	MOVWF SAYAC
	RETFIE



	END
