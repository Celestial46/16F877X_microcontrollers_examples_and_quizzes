#INCLUDE <P16F877A.INC>
SAYI EQU 0X20
SAYAC1 EQU 0X21
SAYAC2 EQU 0X22

	ORG 0X00
	GOTO MAIN
	ORG 0X04
	ORG 0X05
MAIN
	BANKSEL TRISB
	CLRF TRISB
	CLRF TRISC
	BANKSEL PORTA
	CLRF PORTB
	CLRF PORTC
	
ISLE
	MOVFW SAYI
	CALL LOOKUPCPORT
	MOVWF PORTC
	MOVFW SAYI
	CALL LOOKUPBPORT
	MOVWF PORTB
	CALL GECIKME
	INCF SAYI,F
	MOVFW SAYI
	SUBLW D'5'
	BTFSS STATUS,Z
	GOTO ISLE
	CLRF SAYI
	GOTO ISLE
		

LOOKUPBPORT
	ADDWF PCL,F
	RETLW B'10000000'
	RETLW B'10110110'
	RETLW B'10110110'
	RETLW B'10110110'
	RETLW B'10110110'
	RETURN
LOOKUPCPORT
	ADDWF PCL,F
	RETLW D'1'
	RETLW D'2'
	RETLW D'4'
	RETLW D'8'
	RETLW D'16'
	RETURN
GECIKME
	MOVLW D'10'
	MOVWF SAYAC1
GECIKME1
	MOVLW D'50'
	MOVWF SAYAC2
GECIKTIR
	DECFSZ SAYAC2,F
	GOTO GECIKTIR
	DECFSZ SAYAC1,F
	GOTO GECIKME1
	RETURN	
END
